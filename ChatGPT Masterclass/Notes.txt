from openai import OpenAI
sec=""

client = OpenAI()
client.api_key=sec
response = client.responses.create(
    model="gpt-5.4",
    input="Write a one-sentence bedtime story about a unicorn.",
	max_tokens=200,
	temperature=0
)

print(response.output_text['choices'][0]['text'])
